USE edutrack_db;

-- 1. PROOF OF SCALE (The 100,000+ rows)
-- Run this first to show the massive dataset size.

SELECT 
    (SELECT COUNT(*) FROM Students) AS Total_Students,
    (SELECT COUNT(*) FROM Enrollments) AS Total_Enrollments,
    (SELECT COUNT(*) FROM Students) + (SELECT COUNT(*) FROM Enrollments) AS GRAND_TOTAL;


-- 2. PROOF OF TRIGGER (Audit Log)
-- Run this AFTER you "Edit a Grade" in the browser. 
-- It shows the trigger automatically captured the history.

SELECT * FROM audit_logs ORDER BY changed_at DESC LIMIT 5;


-- 3. PROOF OF INDEXING (Performance Benchmark)
-- Run this to show that search is fast because of your work in Phase 5.
-- Point to the 'type: ref' result.

EXPLAIN SELECT * FROM Students WHERE first_name = 'Holly';


-- 4. PROOF OF MANY-TO-MANY (Junction Table)
-- Show how Students are linked to Courses through Enrollments.

SELECT s.first_name, s.last_name, c.course_title, e.grade
FROM Students s
JOIN Enrollments e ON s.student_id = e.student_id
JOIN Courses c ON e.course_id = c.course_id
LIMIT 5;


-- 5. BUSINESS INTELLIGENCE (The "Manager" View)
-- Show which department has the most students.

SELECT d.dept_name, COUNT(e.enrollment_id) as student_count
FROM Enrollments e
JOIN Courses c ON e.course_id = c.course_id
JOIN Departments d ON c.dept_id = d.dept_id
GROUP BY d.dept_name
ORDER BY student_count DESC;
